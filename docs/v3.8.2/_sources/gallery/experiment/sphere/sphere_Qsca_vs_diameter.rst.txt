.. code-block:: Python

    source = Gaussian(
        wavelength=[405, 810] * ureg.nanometer,
        polarization=0 * ureg.degree,
        optical_power=1e-3 * ureg.watt,
        NA=0.2 * ureg.AU
    )

    scatterer = Sphere(
        diameter=np.linspace(10, 1000, 150) * ureg.nanometer,
        medium_property=[1.33] * ureg.RIU,
        property=Material.polystyren,
        source=source
    )
